 392:      *
 393:      * @param int|null $numBytes
 394:      *    You can pass in <code>null</code>.  But if you know how many bytes you expect, pass in
 395:      *    that value and this function will do a sanity check at the end to make sure the number of
 396:      *    bytes read from $inStream matches up.
 397:      *
 398:      * @param int $chunkSize
 399:      *
 400:      * @return array
 401:      *    The <a href="https://www.dropbox.com/developers/core/docs#metadata-details>metadata
 402:      *    object</a> for the newly-added file.
 403:      */
 404:     private function _uploadFileChunked($path, $writeMode, $inStream, $numBytes, $chunkSize)
 405:     {
 406:         Path::checkArg("path", $path);
 407:         WriteMode::checkArg("writeMode", $writeMode);
 408:         Checker::argResource("inStream", $inStream);
 409:         Checker::argNatOrNull("numBytes", $numBytes);
 410:         Checker::argNat("chunkSize", $chunkSize);
 411: 
 412:         // NOTE: This function performs 3 retries on every call.  This is maybe not the right
 413:         // layer to make retry decisions.  It's also awkward because none of the other calls
 414:         // perform retries.
 415: 
 416:         assert($chunkSize > 0);
 417: 
 418:         $data = self::readFully($inStream, $chunkSize);
 419:         $len = strlen($data);
 420: 
 421:         $client = $this;
 422:         $uploadId = RequestUtil::runWithRetry(3, function() use ($data, $client) {
 423:             return $client->chunkedUploadStart($data);
 424:         });
 425: 
 426:         $byteOffset = $len;
 427: 
 428:         while (!feof($inStream)) {
 429:             $data = self::readFully($inStream, $chunkSize);
 430:             $len = strlen($data);
 431: 
 432:             while (true) {
 433:                 $r = RequestUtil::runWithRetry(3,
 434:                     function() use ($client, $uploadId, $byteOffset, $data) {
 435:                         return $client->chunkedUploadContinue($uploadId, $byteOffset, $data);
 436:                     });
 437: 
 438:                 if ($r === true) {  // Chunk got uploaded!
 439:                     $byteOffset += $len;
 440:                     break;
 441:                 }
 442:                 if ($r === false) {  // Server didn't recognize our upload ID
 443:                     // This is very unlikely since we're uploading all the chunks in sequence.
 444:                     throw new Exception_BadResponse("Server forgot our uploadId");
 445:                 }
 446: 
 447:                 // Otherwise, the server is at a different byte offset from us.
 448:                 $serverByteOffset = $r;
 449:                 assert($serverByteOffset !== $byteOffset);  // chunkedUploadContinue ensures this.
 450:                 // An earlier byte offset means the server has lost data we sent earlier.
 451:                 if ($serverByteOffset < $byteOffset) throw new Exception_BadResponse(
 452:                     "Server is at an ealier byte offset: us=$byteOffset, server=$serverByteOffset");
 453:                 $diff = $serverByteOffset - $byteOffset;
 454:                 // If the server is past where we think it could possibly be, something went wrong.
 455:                 if ($diff > $len) throw new Exception_BadResponse(
 456:                     "Server is more than a chunk ahead: us=$byteOffset, server=$serverByteOffset");
 457:                 // The normal case is that the server is a bit further along than us because of a
 458:                 // partially-uploaded chunk.  Finish it off.
 459:                 $byteOffset += $diff;
 460:                 if ($diff === $len) break;  // If the server is at the end, we're done.
 461:                 $data = substr($data, $diff);
 462:             }
 463:         }
 464: 
 465:         if ($numBytes !== null && $byteOffset !== $numBytes) throw new \InvalidArgumentException(
 466:             "You passed numBytes=$numBytes but the stream had $byteOffset bytes.");
 467: 
 468:         $metadata = RequestUtil::runWithRetry(3,
 469:             function() use ($client, $uploadId, $path, $writeMode) {
 470:                 return $client->chunkedUploadFinish($uploadId, $path, $writeMode);
 471:             });
 472: 
 473:         return $metadata;
 474:     }

 476:     /**
 477:      * Sometimes fread() returns less than the request number of bytes (for example, when reading
 478:      * from network streams).  This function repeatedly calls fread until the requested number of
 479:      * bytes have been read or we've reached EOF.
 480:      *
 481:      * @param resource $inStream
 482:      * @param int $numBytes
 483:      * @throws StreamReadException
 484:      * @return string
 485:      */
 486:     private static function readFully($inStream, $numBytes)
 487:     {
 488:         Checker::argNat("numBytes", $numBytes);
 489: 
 490:         $full = '';
 491:         $bytesRemaining = $numBytes;
 492:         while (!feof($inStream) && $bytesRemaining > 0) {
 493:             $part = fread($inStream, $bytesRemaining);
 494:             if ($part === false) throw new StreamReadException("Error reading from \$inStream.");
 495:             $full .= $part;
 496:             $bytesRemaining -= strlen($part);
 497:         }
 498:         return $full;
 499:     }

 571:      * @param string $uploadId
 572:      *     The unique identifier for the chunked upload session.  This is obtained via
 573:      *     {@link chunkedUploadStart}.
 574:      *
 575:      * @param int $byteOffset
 576:      *     The number of bytes you think you've already uploaded to the given chunked upload
 577:      *     session.  The server will append the new chunk of data after that point.
 578:      *
 579:      * @param string $data
 580:      *     The data to append to the existing chunked upload session.
 581:      *
 582:      * @return int|bool
 583:      *     If <code>false</code>, it means the server didn't know about the given
 584:      *     <code>$uploadId</code>.  This may be because the chunked upload session has expired
 585:      *     (they last around 24 hours).
 586:      *     If <code>true</code>, the chunk was successfully uploaded.  If an integer, it means
 587:      *     you and the server don't agree on the current <code>$byteOffset</code>.  The returned
 588:      *     integer is the server's internal byte offset for the chunked upload session.  You need
 589:      *     to adjust your input to match.
 590:      *
 591:      * @throws Exception
 592:      */
 593:     function chunkedUploadContinue($uploadId, $byteOffset, $data)
 594:     {
 595:         Checker::argStringNonEmpty("uploadId", $uploadId);
 596:         Checker::argNat("byteOffset", $byteOffset);
 597:         Checker::argString("data", $data);
 598: 
 599:         $response = $this->_chunkedUpload(
 600:             array("upload_id" => $uploadId, "offset" => $byteOffset), $data);
 601: 
 602:         if ($response->statusCode === 404) {
 603:             // The server doesn't know our upload ID.  Maybe it expired?
 604:             return false;
 605:         }
 606: 
 607:         $correction = self::_chunkedUploadCheckForOffsetCorrection($response);
 608:         if ($correction !== null) {
 609:             list($correctedUploadId, $correctedByteOffset) = $correction;
 610:             if ($correctedUploadId !== $uploadId) throw new Exception_BadResponse(
 611:                 "Corrective 400 upload_id mismatch: us=".
 612:                 self::q($uploadId)." server=".self::q($correctedUploadId));
 613:             if ($correctedByteOffset === $byteOffset) throw new Exception_BadResponse(
 614:                 "Corrective 400 offset is the same as ours: $byteOffset");
 615:             return $correctedByteOffset;
 616:         }
 617: 
 618:         if ($response->statusCode !== 200) throw RequestUtil::unexpectedStatus($response);
 619:         list($retUploadId, $retByteOffset) = self::_chunkedUploadParse200Response($response->body);
 620: 
 621:         $nextByteOffset = $byteOffset + strlen($data);
 622:         if ($uploadId !== $retUploadId) throw new Exception_BadResponse(
 623:                 "upload_id mismatch: us=".self::q($uploadId).", server=".self::q($uploadId));
 624:         if ($nextByteOffset !== $retByteOffset) throw new Exception_BadResponse(
 625:                 "next-offset mismatch: us=$nextByteOffset, server=$retByteOffset");
 626: 
 627:         return true;
 628:     }
